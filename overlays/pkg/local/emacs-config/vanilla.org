#+TITLE: Vanilla Emacs Configuration
#+AUTHOR: Yann
#+PROPERTY: header-args:emacs-lisp :noweb yes :results none :tangle no

Vanilla Emacs configuration managed by Nix. Packages are installed via
Nix (=:ensure nil= everywhere), use-package handles lazy loading, and
common blocks from =common.org= are shared with the Doom config.

* Early Init
:PROPERTIES:
:header-args:emacs-lisp: :tangle early-init.el :comments no :noweb yes
:END:

#+begin_src emacs-lisp
;;; early-init.el --- Early init -*- lexical-binding: t -*-

;; Increase GC threshold during startup for faster loading
(setq gc-cons-threshold most-positive-fixnum
      gc-cons-percentage 0.6)

;; Disable package.el — Nix manages all packages
(setq package-enable-at-startup nil)

;; Prevent the glimpse of un-styled Emacs by disabling UI elements early
(push '(menu-bar-lines . 0) default-frame-alist)
(push '(tool-bar-lines . 0) default-frame-alist)
(push '(vertical-scroll-bars) default-frame-alist)

;; Prevent resize flicker
(setq frame-inhibit-implied-resize t)

;; Native compilation settings
(when (featurep 'native-compile)
  (setq native-comp-async-report-warnings-errors 'silent
        native-comp-deferred-compilation t))
#+end_src

** Preamble

#+begin_src emacs-lisp
;;; init.el --- Main init -*- lexical-binding: t -*-

;; Reset GC threshold after startup
(add-hook 'emacs-startup-hook
          (lambda ()
            (setq gc-cons-threshold (* 16 1024 1024)
                  gc-cons-percentage 0.1)))
#+end_src

** Identity

The =+id.el= file is generated by Nix based on machine context.

#+begin_src emacs-lisp
(let ((id-file (expand-file-name "+id.el" user-emacs-directory)))
  (when (file-exists-p id-file)
    (load id-file)))
#+end_src

** Custom file

#+begin_src emacs-lisp
(setq custom-file (expand-file-name "custom.el" user-emacs-directory))
(when (file-exists-p custom-file)
  (load custom-file))
#+end_src

** use-package defaults

All packages come from Nix, so disable automatic installation.

#+begin_src emacs-lisp
(setq use-package-always-ensure nil)
#+end_src

** Core settings

#+begin_src emacs-lisp
;; Encoding
(set-language-environment "UTF-8")
(prefer-coding-system 'utf-8)

;; Recent files
(use-package recentf
  :init
  (recentf-mode 1)
  :config
  (setq recentf-max-saved-items 200))

;; Save minibuffer history
(use-package savehist
  :init
  (savehist-mode 1))

;; Remember cursor position
(use-package saveplace
  :init
  (save-place-mode 1))

;; Auto-revert files changed on disk
(use-package autorevert
  :init
  (global-auto-revert-mode 1)
  :config
  (setq global-auto-revert-non-file-buffers t))

;; Unique buffer names
(use-package uniquify
  :config
  (setq uniquify-buffer-name-style 'forward))

;; Auto-close brackets
(use-package elec-pair
  :init
  (electric-pair-mode 1))

;; Short answers
(setq use-short-answers t)

;; Don't litter with backup/lock files
(setq create-lockfiles nil
      make-backup-files nil
      auto-save-default t
      auto-save-file-name-transforms `((".*" ,temporary-file-directory t)))

;; Scrolling
(setq scroll-margin 2
      scroll-conservatively 101
      scroll-preserve-screen-position t)

;; Indentation
(setq-default indent-tabs-mode nil
              tab-width 4)
#+end_src

** UI

#+begin_src emacs-lisp
;; Theme
(use-package doom-themes
  :config
  (load-theme 'doom-tomorrow-night t)
  (doom-themes-org-config))

;; Modeline
(use-package doom-modeline
  :init
  (doom-modeline-mode 1)
  :config
  (setq doom-modeline-height 28))

;; Font
(set-face-attribute 'default nil
                    :family "FiraCode Nerd Font"
                    :height 130
                    :weight 'regular)
(set-face-attribute 'variable-pitch nil
                    :family "FiraCode Nerd Font"
                    :height 130)

;; Disable line numbers (match Doom config)
(setq display-line-numbers-type nil)

;; Icons
(use-package nerd-icons)

;; Highlight TODOs
(use-package hl-todo
  :hook (prog-mode . hl-todo-mode))

;; VCS diff in fringe
(use-package diff-hl
  :hook ((prog-mode . diff-hl-mode)
         (dired-mode . diff-hl-dired-mode)
         (magit-post-refresh . diff-hl-magit-post-refresh)))

;; Which-key for keybinding discovery
(use-package which-key
  :init
  (which-key-mode 1)
  :config
  (setq which-key-idle-delay 0.5))
#+end_src
