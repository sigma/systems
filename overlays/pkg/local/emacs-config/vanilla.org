#+TITLE: Vanilla Emacs Configuration
#+AUTHOR: Yann
#+PROPERTY: header-args:emacs-lisp :noweb yes :results none :tangle no

Vanilla Emacs configuration managed by Nix. Packages are installed via
Nix (=:ensure nil= everywhere), use-package handles lazy loading, and
common blocks from =common.org= are shared with the Doom config.

* Early Init
:PROPERTIES:
:header-args:emacs-lisp: :tangle early-init.el :comments no :noweb yes
:END:

#+begin_src emacs-lisp
;;; early-init.el --- Early init -*- lexical-binding: t -*-

;; Increase GC threshold during startup for faster loading
(setq gc-cons-threshold most-positive-fixnum
      gc-cons-percentage 0.6)

;; Let Nix wrapper's package initialization load autoloads
;; (package.el won't install anything — Nix manages all packages)

;; Prevent the glimpse of un-styled Emacs by disabling UI elements early
(push '(menu-bar-lines . 0) default-frame-alist)
(push '(tool-bar-lines . 0) default-frame-alist)
(push '(vertical-scroll-bars) default-frame-alist)

;; Prevent resize flicker
(setq frame-inhibit-implied-resize t)

;; Native compilation settings
(when (featurep 'native-compile)
  (setq native-comp-async-report-warnings-errors 'silent
        native-comp-deferred-compilation t))
#+end_src

** Identity

The =+id.el= file is generated by Nix based on machine context.

#+begin_src emacs-lisp
(let ((id-file (expand-file-name "+id.el" user-emacs-directory)))
  (when (file-exists-p id-file)
    (load id-file)))

<<bibtex-parsing-functions>>

(use-package org-roam-bibtex
  :after org-roam
  :config
  (setq orb-roam-ref-format 'org-cite)
  (add-hook 'org-roam-mode-hook (lambda () (org-roam-bibtex-mode 1))))

(with-eval-after-load 'bibtex
  <<bibtex-dialect-settings>>)

(with-eval-after-load 'org
  <<org-bibtex-types-config>>)

(with-eval-after-load 'org
  (let* ((dir (file-name-as-directory (expand-file-name org-directory)))
         (bibfiles (mapcar
                    (lambda (f) (concat dir f))
                    (directory-files dir nil "\\.bib"))))
    (with-eval-after-load 'citar
      (setq citar-bibliography bibfiles)
      (setq citar-notes-paths (list (file-name-as-directory (expand-file-name "bib" org-roam-directory)))))
    (with-eval-after-load 'bibtex-completion
      (setq bibtex-completion-bibliography bibfiles
            bibtex-completion-notes-path (concat (file-name-as-directory org-roam-directory) "notes")
            bibtex-completion-additional-search-fields '(keywords)
            bibtex-completion-pdf-open-function
            (lambda (fpath)
              (call-process "open" nil 0 nil fpath))))))

(use-package citar-org-roam
  :after (citar org-roam)
  :config
  <<citar-org-roam-settings>>)
